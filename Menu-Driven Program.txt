#include <stdio.h>

int main() {
    int choice;
    int a, b;

    printf("Menu:\n");
    printf("1. Sum\n");
    printf("2. Difference\n");
    printf("3. Product\n");
    printf("4. Division\n");
    printf("5. Modulus\n");
    printf("Enter your choice: ");
    scanf("%d", &choice);
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);

    switch(choice) {
        case 1:
            printf("Sum = %d\n", a + b);
            break;
        case 2:
            printf("Difference = %d\n", a - b);
            break;
        case 3:
            printf("Product = %d\n", a * b);
            break;
        case 4:
            if(b != 0)
                printf("Division = %.d\n", a / b);
            else
                printf("Error: Division by zero\n");
            break;
        case 5:
            if(b != 0)
                printf("Modulus = %d\n", a % b);
            else
                printf("Error: Modulo by zero\n");
            break;
        default:
            printf("Invalid choice\n");
    }

    return 0;
}
