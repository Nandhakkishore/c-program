#include <stdio.h>

int main() {
    char ch;
    printf("Input: ");
    scanf("%c", &ch);
    if ((ch >= 65 && ch <= 90) || (ch >= 97 && ch <= 122)) {
        printf("Output: Alphabet\n");
    }
    else if (ch >= 48 && ch <= 57) {
        printf("Output: Digit\n");
    }
    else {
        printf("Output: Special Character\n");
    }

    return 0;
}
